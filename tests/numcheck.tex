\documentclass{scrartcl}
\usepackage[latin1]{inputenc}
\usepackage[T1]{fontenc}
\usepackage[ngerman]{babel}
\usepackage{lmodern}
\newif\ifinteger
% \gobm{...} gobbles a minus sign
\newcommand{\gobm}[1]{\ifx-#1\expandafter\gobm\else#1\fi}
% \checkifinteger{...} checks if its parameter is an integer
\newcommand{\checkifinteger}[1]{%
  \ifcat _\ifnum9<1\gobm{#1} _\else A\fi
    \integertrue  % an integer
  \else
    \integerfalse % not an integer
  \fi} 
\begin{document}
\def\blubb{a}
\checkifinteger\blubb
\ifinteger
   integer
\else
   not an integer
\fi
\end{document}
