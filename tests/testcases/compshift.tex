\begin{document}
\begin{pspicture}[showgrid=true](10,10)
  \pnode(0,1){A}\pnode(5,1){B}
  \lens[compshift=0.3](A)(B){Lens}
  \drawbeam[beamangleabs]{(A)}{1}{(B)}.
  \pnode(0,3){C}\pnode(5,5){D}\pnode(3,2){E}
  \psdot(D)
  \mirror[compshift=0.3](C)(D)(E){Mirror}
  \drawbeam{(C)}{2}{(E)}
\end{pspicture}
